IF ELSE

if (condition) {
code to be executed if this condition is true;
} elseif (condition) {
code to be executed if first condition is false and this condition is true;
} else {
code to be executed if all conditions are false;
}

SWITCH

switch (n) {
case label1:
code to be executed if n=label1;
case label2:
code to be executed if n=label2;
...
default:
code to be executed if n is different from all labels; } 


        $hewan = 'Kucing';
        
            switch($hewan){
                case 'Anjing':
                    echo 'Ini adalah anjing';
                    case 'Ayam':
                        echo 'Ini adalah ayam';
                        default:
                            echo 'Ini adalah kucing';
            }
$hewan bernilai Kucing.Masuk pada seleksi kondisi switch dengan kondisi hewan, pada line 5 jika nilai hewan
adalah Anjing maka cetak Ini adalah anjing.
Pada line 8 jika nilai hewan adalah Ayam maka cetak Ini adalah ayam.
Pada line 10 jika tidak memenuhi kondisi semuanya maka cetak Ini adalah kucing.
