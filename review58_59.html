unset(), isset()

unset()

unset() digunakan untuk menghapus isi

dari sebuah variabel

unsetTest1
$name = "James";
unset($name);
echo $name;
Error! akan mengeluarkan PHP Notice

isset()

isset() digunakan untuk mengetahui apakah

sebuah variabel sudah memiliki isi atau belum

fungsi ini mengembalikan nilai TRUE dan FALSE

issetTest2
$name = "Michael";
$s1 = isset($name);
akan mengeluarkan TRUE
issetTest
$name = null;
$s1 = isset($name);
akan mengeluarkan FALSE
Testing1
?php 
$name = null;
if(!isset(name)){
 echo "Name is not set";
}
else{
 echo "Name is set";
}
Output = Name is not set
Testing2
?php 
$name = "James";
if(!isset(name)){
 echo "Name is not set";
}
else{
 echo "Name is set to $name";
}
Output = Name is set to James
