File konversi.php

<?php
  if ($_SERVER['REQUEST_METHOD'] == 'POST'):
    if (isset($_POST['jumlah'])):
      $jumlah = $_POST['jumlah'];
        
      for ($i=0; $i<$jumlah; $i++):
        echo
          '<form method="POST">
            Nilai ke-'.($i+1).':
            <input type="number" name="'."nilai$i".'" min="0" max="4" step="0.01" required><br>
          ';
      endfor;
      echo '<input type="submit" value="Konversi nilai"></form>';
    else:
      foreach($_POST as $nilai):
        $out = "Konversi nilai $nilai ke huruf adalah ";

        if ($nilai >= 3.75):
          $out .= "A";
        elseif ($nilai >= 3.25):
          $out .= "B+";
        elseif ($nilai >= 2.75):
          $out .= "B";
        elseif ($nilai >= 2.25):
          $out .= "C+";
        elseif ($nilai >= 1.75):
          $out .= "C";
        elseif ($nilai >= 1.25):
          $out .= "D";
        else:
          $out .= "E";
        endif;
        
        echo "<p>$out</p>";
      endforeach;
    endif;
        
  endif;
  ?>
