AJAX POP-UP

AJAX merupakan singkatan dari Asynchronous JavaScript and XML yang memungkinkan adanya komunikasi dengan server tanpa harus memuat halaman.

Berikut ini adalah contoh kode untuk membuat AJAX POP-UP.

Ajax.html


    <label>
        Name: <input type="text" id="name">
    </label>
    <button id="greet">Greet</button>
    <script src="Ajax.js"></script>

Ajax.js


    const button = document.querySelector('#greet');

    button.addEventListener('click', () => {
                    
        const name = document.querySelector('#name').value;
        const url = 'http://localhost/AJAX/Ajax.php';
                    
        fetch(url, {
            method: 'post',
            headers: {
                'Content-Type': 'application/x-www-form-urlencoded',
            },
                body: `name=${name}`
            })
                .then(res => res.text())
                .then(res => {
                    alert(res);
                });
        });

Ajax.php


    <?php
    sleep(1);
    $name = $_POST['name'];
    echo "Hello $name!";
